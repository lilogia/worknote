Oracle SQL script to create schema before install EM/Systinet

HPe EM/Systinet ››

DEFINE us = &username;

DEFINE pass = "changeit";


CREATE USER &&us IDENTIFIED BY &&pass ;

GRANT SELECT ON sys.dba_pending_transactions TO &&us;

GRANT SELECT ON sys.pending_trans$ TO &&us;

GRANT SELECT ON sys.dba_2pc_pending TO &&us;

GRANT EXECUTE ON sys.dbms_system TO &&us;

GRANT EXECUTE ON sys.dbms_xa TO &&us;

GRANT CONNECT, RESOURCE TO &&us WITH ADMIN OPTION;

GRANT CREATE ANY VIEW, CREATE ANY MATERIALIZED VIEW TO &&us;


/* FOR ORACLE 12C GRANT SOME MORE EXPLICIT PRIVILEGES  */


GRANT Unlimited tablespace  TO &&us;

GRANT CREATE SESSION, CREATE PROCEDURE, CREATE SEQUENCE TO &&us;
